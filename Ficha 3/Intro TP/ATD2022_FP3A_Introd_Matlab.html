%---Verifica a existncia de NaN
disp('H NaN?')
haNaN_IT=any(isnan(x_IT11)) % H NaN por colunas?

% - Reconstruir as linhas eliminadas usando extrapolao (interp1)
x_IT11r=x_IT11;
if haNaN_IT
    ind = indNaN_IT;
    for k=1:length(ind)
        nn=n_IT11(ind(k)-4:ind(k)-1); % admitindo que no h NaNs no incio
        xx=x_IT11r(ind(k)-4:ind(k)-1);
        x_IT11r(ind(k))=interp1(nn,xx,n_IT11(ind(k)),'pchip','extrap');

% Ex 1.5
%---Verifica outliers

mu_PT26_tend = mean(x_PT26_tend) % mdia
sigma_PT26_tend = std(x_PT26_tend) % desvio padro

indoutl = find(abs(x_PT26_tend - mu_PT26_tend) > 3*sigma_PT26_tend) % Identifica outliers
noutl = length(indoutl) % nmero de outliers
x_PT26_tend(indoutl)

x_PT26_tend_o=x_PT26_tend; % Substituio dos outliers
if noutl
    for k=1:numel(indoutl)
        if x_PT26_tend(indoutl(k)) > mu_PT26_tend
            x_PT26_tend_o(indoutl(k)) = mu_PT26_tend + 0.1*sigma_PT26_tend;
        else
            x_PT26_tend_o(indoutl(k)) = mu_PT26_tend - 0.1*sigma_PT26_tend;
        end
    end
end

x_PT26_o=x_PT26_tend_o + tend_x_PT26; % srie original sem outliers

figure(5) % representao grfica dos resultados
subplot(211)
plot(n_PT26,x_PT26_tend,'-+',n_PT26,x_PT26_tend_o,'-d');
legend('Srie temporal sem tendncia','Srie sem tendncia e sem outliers','Location','northwest')
